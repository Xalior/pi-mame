# OUR card boot config — a single-machine card. The Raspberry Pi firmware reads
# this at power-on and boots the kernel named for the board it finds itself on.
#
# Two vocabularies meet here, and they are NOT the same word:
#   - the [pi4] SECTION name is the Foundation firmware's board filter. It is
#     fixed by the boot firmware (the firmware chooses which section applies) —
#     not ours to rename.
#   - the `rpi4` in the kernel FILENAME (kernel-rpi4.img) is our on-card board
#     token, matching the per-board build tree (build/rpi4) and — on a platform
#     card — the generic name the picker chain-boots.
#
# A single-machine card has no picker: the firmware boots the MAME core
# (kernel-<board>.img) directly.
#
# This is a multi-section router, but pi-mame ships PER-BOARD cards: a given
# card carries only its own board's firmware and image, so only its section
# ever resolves. The other sections are inert (their kernels are not on the
# card) and simply document the full board routing.

arm_64bit=1
kernel_address=0x80000
initial_turbo=0

# --- RASPPI 3 world (BCM2837): kernel-rpi3.img ---
[pi02]
kernel=kernel-rpi3.img
[pi2]
kernel=kernel-rpi3.img
[pi3]

[pi3+]
kernel=kernel-rpi3.img
[cm3]
kernel=kernel-rpi3.img
