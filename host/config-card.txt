# OUR card boot config — a platform card. The Raspberry Pi firmware reads this
# at power-on and boots the kernel named for the board it finds itself on.
#
# Two vocabularies meet here, and they are NOT the same word:
#   - the [pi4] SECTION name is the Foundation firmware's board filter. It is
#     fixed by the boot firmware (the firmware chooses which section applies) —
#     not ours to rename.
#   - the `rpi4` in the kernel FILENAME is Circle's image-suffix vocabulary
#     (Circle's own images are kernel8-rpi4.img), which also reads as "RPi".
#     It is the on-card board token, matching the picker's compile-time
#     SYSTEMBIT (boot-picker/kernel.h).
#
# On a platform card the firmware boots the picker, which then chain-boots the
# MAME core (pi-mame-core-rpi4.img).
#
# PoC3 adds a [pi5] section here pointing at that board's kernel; a section
# fans multiple firmware filters (e.g. [pi02], [pi3], [pi3+]) into one Circle
# build-family core as boards are added.

arm_64bit=1
kernel_address=0x80000
initial_turbo=0

[pi4]
armstub=armstub8-rpi4.bin
kernel=pi-mame-boot-rpi4.img
max_framebuffers=2
