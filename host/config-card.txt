# OUR card boot config — a platform card. The Raspberry Pi firmware reads this
# at power-on and boots the picker image named for the board it finds itself on.
#
# Two vocabularies meet here, and they are NOT the same word:
#   - the [pi4] SECTION name is the Foundation firmware's board filter. It is
#     fixed by the boot firmware (the firmware chooses which section applies) —
#     not ours to rename.
#   - the `rpi4` in the picker FILENAME (pi-mame-boot-rpi4.img) is our on-card
#     board token, matching the payload named kernel-rpi4.img the picker then
#     chain-boots and the per-board build tree (build/rpi4).
#
# On a platform card the firmware boots the picker (pi-mame-boot-<board>.img),
# which reads bootmenu.cfg and chain-boots the MAME core (kernel-<board>.img —
# the generic name the picker bakes; the generic bootloader carries no pi-mame
# references, so its payload is named by board alone).
#
# This is a multi-section router, but pi-mame ships PER-BOARD cards: a given
# card carries only its own board's firmware and images, so only its section
# ever resolves. The other sections are inert (their kernels are not on the
# card) and simply document the full board routing.

arm_64bit=1
kernel_address=0x80000
initial_turbo=0

# --- RASPPI 3 world (BCM2837): pi-mame-boot-rpi3.img ---
[pi02]
kernel=pi-mame-boot-rpi3.img
[pi2]
kernel=pi-mame-boot-rpi3.img
[pi3]
kernel=pi-mame-boot-rpi3.img
[pi3+]
kernel=pi-mame-boot-rpi3.img
[cm3]
kernel=pi-mame-boot-rpi3.img

max_framebuffers=2
